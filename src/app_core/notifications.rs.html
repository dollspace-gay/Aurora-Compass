56/// Types of notifications
57#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
58#[serde(rename_all = "kebab-case")]
59pub enum NotificationType {
60    /// Someone liked your post
61    PostLike,
62    /// Someone reposted your post
63    Repost,
64    /// Someone mentioned you
65    Mention,
66    /// Someone replied to your post
67    Reply,
68    /// Someone quoted your post
69    Quote,
70    /// Someone followed you
71    Follow,
72    /// Someone liked your feed generator
73    FeedgenLike,
74    /// Someone joined a starter pack
75    StarterpackJoined,
76    /// Account verified
77    Verified,
78    /// Account unverified
79    Unverified,
80    /// Like via repost
81    LikeViaRepost,
82    /// Repost via repost
83    RepostViaRepost,
84    /// Subscribed post
85    SubscribedPost,
86    /// Unknown notification type
87    Unknown,
88}
89
90impl NotificationType {
91    /// Check if this notification type can be grouped with others
92    pub fn is_groupable(&self) -> bool {
93        GROUPABLE_REASONS.contains(&self.to_reason())
94    }
95
96    /// Convert from AT Protocol reason string
97    pub fn from_reason(reason: &str, reason_subject: Option<&str>) -> Self {
98        match reason {
99            "like" => {
100                if reason_subject
101                    .map(|s| s.contains("feed.generator"))
102                    .unwrap_or(false)
103                {
104                    NotificationType::FeedgenLike
105                } else {
106                    NotificationType::PostLike
107                }
108            }
109            "repost" => NotificationType::Repost,
110            "mention" => NotificationType::Mention,
111            "reply" => NotificationType::Reply,
112            "quote" => NotificationType::Quote,
113            "follow" => NotificationType::Follow,
114            "starterpack-joined" => NotificationType::StarterpackJoined,
115            "verified" => NotificationType::Verified,
116            "unverified" => NotificationType::Unverified,
117            "like-via-repost" => NotificationType::LikeViaRepost,
118            "repost-via-repost" => NotificationType::RepostViaRepost,
119            "subscribed-post" => NotificationType::SubscribedPost,
120            _ => NotificationType::Unknown,
121        }
122    }
123
124    /// Convert to AT Protocol reason string
125    pub fn to_reason(&self) -> &'static str {
126        match self {
127            NotificationType::PostLike => "like",
128            NotificationType::Repost => "repost",
129            NotificationType::Mention => "mention",
130            NotificationType::Reply => "reply",
131            NotificationType::Quote => "quote",
132            NotificationType::Follow => "follow",
133            NotificationType::FeedgenLike => "like",
134            NotificationType::StarterpackJoined => "starterpack-joined",
135            NotificationType::Verified => "verified",
136            NotificationType::Unverified => "unverified",
137            NotificationType::LikeViaRepost => "like-via-repost",
138            NotificationType::RepostViaRepost => "repost-via-repost",
139            NotificationType::SubscribedPost => "subscribed-post",
140            NotificationType::Unknown => "unknown",
141        }
142    }
143}

179impl Notification {
180    /// Get the notification type
181    pub fn notification_type(&self) -> NotificationType {
182        NotificationType::from_reason(&self.reason, self.reason_subject.as_deref())
183    }
184
185    /// Get the subject URI for this notification
186    pub fn subject_uri(&self) -> Option<String> {
187        let notif_type = self.notification_type();
188        match notif_type {
189            NotificationType::Reply
190            | NotificationType::Quote
191            | NotificationType::Mention
192            | NotificationType::SubscribedPost => Some(self.uri.clone()),
193            NotificationType::PostLike
194            | NotificationType::Repost
195            | NotificationType::LikeViaRepost
196            | NotificationType::RepostViaRepost => {
197                // Extract subject URI from record
198                self.record
199                    .get("subject")
200                    .and_then(|s| s.get("uri"))
201                    .and_then(|u| u.as_str())
202                    .map(String::from)
203            }
204            NotificationType::FeedgenLike => self.reason_subject.clone(),
205            _ => None,
206        }
207    }
208
209    /// Parse the indexed_at timestamp to milliseconds since epoch
210    pub fn indexed_at_ms(&self) -> i64 {
211        chrono::DateTime::parse_from_rfc3339(&self.indexed_at)
212            .map(|dt| dt.timestamp_millis())
213            .unwrap_or(0)
214    }
215}

286impl FeedNotification {
287    /// Create a new feed notification from a raw notification
288    pub fn new(notification: Notification) -> Self {
289        let notification_type = notification.notification_type();
290        let subject_uri = notification.subject_uri();
291        let react_key = format!("notif-{}-{}", notification.uri, notification.reason);
292
293        FeedNotification {
294            react_key,
295            notification_type,
296            notification,
297            additional: None,
298            subject_uri,
299            subject: None,
300        }
301    }
302
303    /// Get total count of notifications in this group
304    pub fn count(&self) -> usize {
305        1 + self.additional.as_ref().map(|a| a.len()).unwrap_or(0)
306    }
307
308    /// Get all authors involved in this notification group
309    pub fn authors(&self) -> Vec<&ProfileViewBasic> {
310        let mut authors = vec![&self.notification.author];
311        if let Some(ref additional) = self.additional {
312            for notif in additional {
313                authors.push(&notif.author);
314            }
315        }
316        authors
317    }
318
319    /// Check if this notification has been read
320    pub fn is_read(&self) -> bool {
321        self.notification.is_read
322    }
323
324    /// Set whether this notification is read based on seen_at time
325    pub fn update_read_status(&mut self, seen_at: &chrono::DateTime<chrono::Utc>) {
326        if let Ok(indexed) = chrono::DateTime::parse_from_rfc3339(&self.notification.indexed_at) {
327            self.notification.is_read = seen_at > &indexed.with_timezone(&chrono::Utc);
328        }
329    }

467/// Groups notifications by reason and subject within a time window
468pub fn group_notifications(notifications: Vec<Notification>) -> Vec<FeedNotification> {
469    let mut grouped: Vec<FeedNotification> = Vec::new();
470
471    for notif in notifications {
472        let ts = notif.indexed_at_ms();
473        let notif_type = notif.notification_type();
474        let mut was_grouped = false;
475
476        // Check if this notification can be grouped
477        if notif_type.is_groupable() {
478            for existing in grouped.iter_mut() {
479                let existing_ts = existing.notification.indexed_at_ms();
480
481                // Check if within time window and same type/subject
482                if (existing_ts - ts).abs() < GROUPING_WINDOW_MS
483                    && notif.reason == existing.notification.reason
484                    && notif.reason_subject == existing.notification.reason_subject
485                {
486                    // Don't group if same author (unless subscribed-post)
487                    let same_author = notif.author.did == existing.notification.author.did;
488                    let is_subscribed = notif.reason == "subscribed-post";
489
490                    if !same_author || is_subscribed {
491                        // Check for follow-back (don't group follow-backs)
492                        let next_is_follow_back = notif.reason == "follow"
493                            && notif
494                                .author
495                                .viewer
496                                .as_ref()
497                                .map(|v| v.following.is_some())
498                                .unwrap_or(false);
499                        let prev_is_follow_back = existing.notification.reason == "follow"
500                            && existing
501                                .notification
502                                .author
503                                .viewer
504                                .as_ref()
505                                .map(|v| v.following.is_some())
506                                .unwrap_or(false);
507
508                        if !next_is_follow_back && !prev_is_follow_back {
509                            // Add to existing group
510                            if existing.additional.is_none() {
511                                existing.additional = Some(Vec::new());
512                            }
513                            existing.additional.as_mut().unwrap().push(notif.clone());
514                            was_grouped = true;
515                            break;
516                        }
517                    }
518                }
519            }
520        }
521
522        if !was_grouped {
523            grouped.push(FeedNotification::new(notif));
524        }
525    }
526
527    grouped
528}

552    /// List notifications with pagination
553    pub async fn list_notifications(
554        &self,
555        params: ListNotificationsParams,
556    ) -> Result<ListNotificationsResponse> {
557        let client = self.client.read().await;
558
559        let mut request = XrpcRequest::query("app.bsky.notification.listNotifications");
560
561        if let Some(limit) = params.limit {
562            request = request.param("limit", limit.to_string());
563        }
564        if let Some(ref cursor) = params.cursor {
565            request = request.param("cursor", cursor.clone());
566        }
567        if let Some(priority) = params.priority {
568            request = request.param("priority", priority.to_string());
569        }
570        if let Some(ref reasons) = params.reasons {
571            for reason in reasons {
572                request = request.param("reasons", reason.clone());
573            }
574        }
575
576        let response = client
577            .query(request)
578            .await
579            .map_err(|e| NotificationError::ApiError(e.to_string()))?;
580
581        let data: ListNotificationsResponse = serde_json::from_value(response.data)?;
582        Ok(data)
583    }

585    /// Fetch a page of notifications with grouping and subject resolution
586    pub async fn fetch_page(
587        &self,
588        cursor: Option<String>,
589        filter: NotificationFilter,
590        fetch_subjects: bool,
591    ) -> Result<NotificationPage> {
592        let params = ListNotificationsParams {
593            limit: Some(PAGE_SIZE),
594            cursor,
595            reasons: filter.reasons(),
596            priority: None,
597        };
598
599        let response = self.list_notifications(params).await?;
600
601        // Group notifications
602        let grouped = group_notifications(response.notifications);
603
604        // Optionally fetch subjects
605        let items = if fetch_subjects {
606            self.resolve_subjects(grouped).await?
607        } else {
608            grouped
609        };
610
611        // Parse seen_at
612        let seen_at = response
613            .seen_at
614            .as_ref()
615            .and_then(|s| chrono::DateTime::parse_from_rfc3339(s).ok())
616            .map(|dt| dt.with_timezone(&chrono::Utc))
617            .unwrap_or_else(chrono::Utc::now);
618
619        Ok(NotificationPage {
620            cursor: response.cursor,
621            seen_at,
622            items,
623            priority: response.priority.unwrap_or(false),
624        })
625    }

627    /// Resolve subjects (posts, starter packs) for notifications
628    async fn resolve_subjects(
629        &self,
630        mut notifications: Vec<FeedNotification>,
631    ) -> Result<Vec<FeedNotification>> {
632        // Collect URIs to fetch
633        let mut post_uris: Vec<String> = Vec::new();
634        let mut pack_uris: Vec<String> = Vec::new();
635
636        for notif in &notifications {
637            if let Some(ref uri) = notif.subject_uri {
638                if uri.contains("app.bsky.feed.post") {
639                    post_uris.push(uri.clone());
640                } else if notif
641                    .notification
642                    .reason_subject
643                    .as_ref()
644                    .map(|s| s.contains("app.bsky.graph.starterpack"))
645                    .unwrap_or(false)
646                {
647                    if let Some(ref reason_subject) = notif.notification.reason_subject {
648                        pack_uris.push(reason_subject.clone());
649                    }
650                }
651            }
652        }
653
654        // Fetch posts in batches of 25
655        let posts = self.fetch_posts(&post_uris).await?;
656        let packs = self.fetch_starter_packs(&pack_uris).await?;
657
658        // Attach subjects to notifications
659        for notif in &mut notifications {
660            if let Some(ref uri) = notif.subject_uri {
661                if notif.notification_type == NotificationType::StarterpackJoined {
662                    if let Some(ref reason_subject) = notif.notification.reason_subject {
663                        if let Some(pack) = packs.get(reason_subject) {
664                            notif.subject =
665                                Some(NotificationSubject::StarterPack(Box::new(pack.clone())));
666                        }
667                    }
668                } else if let Some(post) = posts.get(uri) {
669                    notif.subject = Some(NotificationSubject::Post(Box::new(post.clone())));
670                }
671            }
672        }
673
674        Ok(notifications)
675    }

677    /// Fetch posts by URIs
678    async fn fetch_posts(&self, uris: &[String]) -> Result<HashMap<String, PostView>> {
679        let mut result = HashMap::new();
680
681        if uris.is_empty() {
682            return Ok(result);
683        }
684
685        let client = self.client.read().await;
686
687        // Fetch in chunks of 25
688        for chunk in uris.chunks(25) {
689            let mut request = XrpcRequest::query("app.bsky.feed.getPosts");
690            for uri in chunk {
691                request = request.param("uris", uri.clone());
692            }
693
694            match client.query::<serde_json::Value>(request).await {
695                Ok(response) => {
696                    if let Some(posts) = response.data.get("posts").and_then(|p| p.as_array()) {
697                        for post_value in posts {
698                            if let Ok(post) = serde_json::from_value::<PostView>(post_value.clone())
699                            {
700                                result.insert(post.uri.clone(), post);
701                            }
702                        }
703                    }
704                }
705                Err(e) => {
706                    // Log error but continue - some posts might be deleted
707                    tracing::warn!("Failed to fetch posts: {}", e);
708                }
709            }
710        }
711
712        Ok(result)
713    }

715    /// Fetch starter packs by URIs
716    async fn fetch_starter_packs(
717        &self,
718        uris: &[String],
719    ) -> Result<HashMap<String, StarterPackViewBasic>> {
720        let mut result = HashMap::new();
721
722        if uris.is_empty() {
723            return Ok(result);
724        }
725
726        let client = self.client.read().await;
727
728        // Fetch in chunks of 25
729        for chunk in uris.chunks(25) {
730            let mut request = XrpcRequest::query("app.bsky.graph.getStarterPacks");
731            for uri in chunk {
732                request = request.param("uris", uri.clone());
733            }
734
735            match client.query::<serde_json::Value>(request).await {
736                Ok(response) => {
737                    if let Some(packs) =
738                        response.data.get("starterPacks").and_then(|p| p.as_array())
739                    {
740                        for pack_value in packs {
741                            if let Ok(pack) =
742                                serde_json::from_value::<StarterPackViewBasic>(pack_value.clone())
743                            {
744                                result.insert(pack.uri.clone(), pack);
745                            }
746                        }
747                    }
748                }
749                Err(e) => {
750                    tracing::warn!("Failed to fetch starter packs: {}", e);
751                }
752            }
753        }

758    /// Get the unread notification count
759    pub async fn get_unread_count(&self) -> Result<u32> {
760        let client = self.client.read().await;
761
762        let request = XrpcRequest::query("app.bsky.notification.getUnreadCount");
763
764        let response = client
765            .query(request)
766            .await
767            .map_err(|e| NotificationError::ApiError(e.to_string()))?;
768
769        let data: UnreadCountResponse = serde_json::from_value(response.data)?;
770
771        // Update cached count
772        let mut count = self.unread_count.write().await;
773        *count = data.count;
774
775        Ok(data.count)
776    }

779    pub async fn mark_all_read(&self) -> Result<()> {
780        let client = self.client.read().await;
781        let now = chrono::Utc::now().to_rfc3339();
782
783        let params = UpdateSeenParams { seen_at: now };
784
785        let request = XrpcRequest::procedure("app.bsky.notification.updateSeen")
786            .json_body(&params)
787            .map_err(|e| NotificationError::ApiError(e.to_string()))?;
788
789        client
790            .procedure::<serde_json::Value>(request)
791            .await
792            .map_err(|e| NotificationError::ApiError(e.to_string()))?;
793
794        // Reset unread count
795        let mut count = self.unread_count.write().await;
796        *count = 0;
797
798        Ok(())
799    }

819    /// Check for new notifications and update cache
820    ///
821    /// If `invalidate` is true, the cached page will be marked as invalid
822    /// and the query will be forced to refetch.
823    pub async fn check_unread(&self, _invalidate: bool) -> Result<u32> {
824        // Fetch first page
825        let page = self.fetch_page(None, NotificationFilter::All, true).await?;
826
827        // Update cache
828        let mut cache = self.cached_page.write().await;
829        cache.synced_at = chrono::Utc::now();
830        cache.usable_in_feed = true;
831
832        // Count unread
833        let unread = page.items.iter().filter(|n| !n.is_read()).count() as u32;
834
835        cache.unread_count = unread;
836        cache.data = Some(page);
837
838        // Update unread count
839        let mut count = self.unread_count.write().await;
840        *count = unread;
841
842        Ok(unread)
843    }

845    /// Get cached page if available and fresh
846    pub async fn get_cached_page(&self) -> Option<NotificationPage> {
847        let cache = self.cached_page.read().await;
848        if cache.usable_in_feed {
849            cache.data.clone()
850        } else {
851            None
852        }
853    }
854
855    /// Get the last known unread count
856    pub async fn last_unread_count(&self) -> u32 {
857        *self.unread_count.read().await
858    }
859
860    /// Invalidate the notification cache
861    pub async fn invalidate_cache(&self) {
862        let mut cache = self.cached_page.write().await;
863        cache.usable_in_feed = false;
864    }
