18/// Errors that can occur during messaging operations
19#[derive(Debug, Error)]
20pub enum MessageError {
21    /// XRPC error
22    #[error("XRPC error: {0}")]
23    Xrpc(String),
24
25    /// Conversation not found
26    #[error("Conversation not found: {0}")]
27    ConversationNotFound(String),
28
29    /// Message not found
30    #[error("Message not found: {0}")]
31    MessageNotFound(String),
32
33    /// Invalid recipient
34    #[error("Invalid recipient: {0}")]
35    InvalidRecipient(String),
36
37    /// Message too long
38    #[error("Message too long: {length} exceeds maximum {max}")]
39    MessageTooLong {
40        /// Actual message length
41        length: usize,
42        /// Maximum allowed length
43        max: usize,
44    },
45
46    /// Serialization error
47    #[error("Serialization error: {0}")]
48    Serialization(#[from] serde_json::Error),
49
50    /// No active session
51    #[error("No active session")]
52    NoSession,
53}

553/// Inbox view for managing conversations
554///
555/// Provides sorting, filtering, and preview generation for conversation lists.
556///
557/// # Example
558///
559/// ```rust
560/// use app_core::messages::{InboxView, Conversation, ConversationSortBy, InboxFilter};
561///
562/// let conversations = vec![/* your conversations */];
563/// let mut inbox = InboxView::new(conversations);
564///
565/// // Sort by most recent
566/// inbox.sort_by(ConversationSortBy::Recent);
567///
568/// // Filter to show only unread
569/// let filter = InboxFilter {
570///     only_unread: true,
571///     ..Default::default()
572/// };
573/// let unread = inbox.filter(&filter);
574///
575/// // Get stats
576/// let stats = inbox.stats();
577/// println!("Total unread: {}", stats.total_unread);
578/// ```
579pub struct InboxView {
580    conversations: Vec<Conversation>,
581}
582
583impl InboxView {
584    /// Create a new inbox view
585    pub fn new(conversations: Vec<Conversation>) -> Self {
586        Self { conversations }
587    }

589    /// Get all conversations
590    pub fn conversations(&self) -> &[Conversation] {
591        &self.conversations
592    }
593
594    /// Get a mutable reference to conversations
595    pub fn conversations_mut(&mut self) -> &mut Vec<Conversation> {
596        &mut self.conversations
597    }
598
599    /// Sort conversations by the specified criterion
600    pub fn sort_by(&mut self, sort_by: ConversationSortBy) {
601        match sort_by {
602            ConversationSortBy::Recent => {
603                self.conversations
604                    .sort_by(|a, b| b.updated_at.cmp(&a.updated_at));
605            }
606            ConversationSortBy::Unread => {
607                self.conversations.sort_by(|a, b| {
608                    // First sort by unread status (unread first)
609                    match (a.has_unread(), b.has_unread()) {
610                        (true, false) => std::cmp::Ordering::Less,
611                        (false, true) => std::cmp::Ordering::Greater,
612                        _ => {
613                            // Then by unread count (descending)
614                            match b.unread_count.cmp(&a.unread_count) {
615                                std::cmp::Ordering::Equal => {
616                                    // Finally by most recent
617                                    b.updated_at.cmp(&a.updated_at)
618                                }
619                                other => other,
620                            }
621                        }
622                    }
623                });
624            }
625            ConversationSortBy::Name => {
626                self.conversations.sort_by(|a, b| {
627                    let name_a = a.other_participant_name().unwrap_or_default().to_lowercase();
628                    let name_b = b.other_participant_name().unwrap_or_default().to_lowercase();
629                    name_a.cmp(&name_b)
630                });
631            }
632        }
633    }
634
635    /// Filter conversations based on criteria
636    pub fn filter(&self, filter: &InboxFilter) -> Vec<&Conversation> {
637        self.conversations
638            .iter()
639            .filter(|conv| {
640                // Filter by unread status
641                if filter.only_unread && !conv.has_unread() {
642                    return false;
643                }
644
645                // Filter by muted status
646                if filter.exclude_muted && conv.muted {
647                    return false;
648                }
649
650                // Filter by search query
651                if let Some(query) = &filter.search_query {
652                    let query_lower = query.to_lowercase();
653
654                    // Search in participant names
655                    let name_match = conv.members.iter().any(|member| {
656                        member
657                            .display_name
658                            .as_ref()
659                            .map(|n| n.to_lowercase().contains(&query_lower))
660                            .unwrap_or(false)
661                            || member.handle.to_lowercase().contains(&query_lower)
662                    });
663
664                    // Search in last message text
665                    let message_match = conv
666                        .last_message
667                        .as_ref()
668                        .map(|msg| msg.text.to_lowercase().contains(&query_lower))
669                        .unwrap_or(false);
670
671                    if !name_match && !message_match {
672                        return false;
673                    }
674                }
675
676                true
677            })
678            .collect()
679    }

681    /// Get inbox statistics
682    pub fn stats(&self) -> InboxStats {
683        let total_conversations = self.conversations.len();
684        let unread_conversations = self
685            .conversations
686            .iter()
687            .filter(|c| c.has_unread())
688            .count();
689        let total_unread = self.conversations.iter().map(|c| c.unread_count).sum();
690
691        InboxStats {
692            total_conversations,
693            unread_conversations,
694            total_unread,
695        }
696    }

721    /// Preview text (last message or empty state)
722    pub preview_text: String,
723    /// Relative timestamp ("2 hours ago", etc.)
724    pub relative_time: String,
725    /// Whether there are unread messages

733impl ConversationPreview {
734    /// Generate a preview from a conversation
735    pub fn from_conversation(conversation: &Conversation) -> Self {
736        let title = if conversation.members.len() == 1 {
737            conversation
738                .other_participant_name()
739                .unwrap_or_else(|| "Unknown".to_string())
740        } else {
741            // Group chat - show comma-separated names
742            conversation
743                .members
744                .iter()
745                .filter_map(|m| {
746                    m.display_name
747                        .as_ref()
748                        .cloned()
749                        .or_else(|| Some(m.handle.clone()))
750                })
751                .take(3)
752                .collect::<Vec<_>>()
753                .join(", ")
754        };
755
756        let preview_text = if let Some(last_msg) = &conversation.last_message {
757            // Truncate long messages
758            const MAX_PREVIEW_LENGTH: usize = 100;
759            if last_msg.text.len() > MAX_PREVIEW_LENGTH {
760                format!("{}...", &last_msg.text[..MAX_PREVIEW_LENGTH])
761            } else {
762                last_msg.text.clone()
763            }
764        } else {
765            "No messages yet".to_string()
766        };
767
768        let relative_time = format_relative_time(&conversation.updated_at);
769
770        Self {
771            conversation_id: conversation.id.clone(),
772            title,
773            preview_text,
774            relative_time,
775            has_unread: conversation.has_unread(),
776            unread_count: conversation.unread_count,
777            is_muted: conversation.muted,
778        }
779    }

783///
784/// Returns strings like "Just now", "5 minutes ago", "2 hours ago", "Yesterday", "3 days ago", etc.
785pub fn format_relative_time(timestamp: &DateTime<Utc>) -> String {
786    let now = Utc::now();
787    let duration = now.signed_duration_since(*timestamp);
788
789    if duration.num_seconds() < 60 {
790        "Just now".to_string()
791    } else if duration.num_minutes() < 60 {
792        let mins = duration.num_minutes();
793        if mins == 1 {
794            "1 minute ago".to_string()
795        } else {
796            format!("{} minutes ago", mins)
797        }
798    } else if duration.num_hours() < 24 {
799        let hours = duration.num_hours();
800        if hours == 1 {
801            "1 hour ago".to_string()
802        } else {
803            format!("{} hours ago", hours)
804        }
805    } else if duration.num_days() == 1 {
806        "Yesterday".to_string()
807    } else if duration.num_days() < 7 {
808        format!("{} days ago", duration.num_days())
809    } else if duration.num_weeks() == 1 {
810        "1 week ago".to_string()
811    } else if duration.num_weeks() < 4 {
812        format!("{} weeks ago", duration.num_weeks())
813    } else {
814        // Show actual date for older messages
815        timestamp.format("%b %d, %Y").to_string()
816    }
817}

931/// Message event listener for real-time updates
932///
933/// Manages connections to receive real-time message events and
934/// distributes them to subscribers via a broadcast channel.
935pub struct MessageEventListener {
936    client: Arc<RwLock<XrpcClient>>,
937    config: EventListenerConfig,
938    event_tx: tokio::sync::broadcast::Sender<MessageEvent>,
939    running: Arc<RwLock<bool>>,
940    last_cursor: Arc<RwLock<Option<String>>>,
941}
942
943impl MessageEventListener {
944    /// Create a new message event listener
945    ///
946    /// # Arguments
947    ///
948    /// * `client` - XRPC client for making requests
949    /// * `config` - Event listener configuration
950    ///
951    /// # Returns
952    ///
953    /// A new `MessageEventListener` instance
954    pub fn new(client: Arc<RwLock<XrpcClient>>, config: EventListenerConfig) -> Self {
955        let (event_tx, _) = tokio::sync::broadcast::channel(config.buffer_size);
956
957        Self {
958            client,
959            config,
960            event_tx,
961            running: Arc::new(RwLock::new(false)),
962            last_cursor: Arc::new(RwLock::new(None)),
963        }
964    }
965
966    /// Subscribe to message events
967    ///
968    /// # Returns
969    ///
970    /// A broadcast receiver for message events
971    pub fn subscribe(&self) -> tokio::sync::broadcast::Receiver<MessageEvent> {
972        self.event_tx.subscribe()
973    }
974
975    /// Check if the listener is currently running
976    ///
977    /// # Returns
978    ///
979    /// True if listener is running, false otherwise
980    pub async fn is_running(&self) -> bool {
981        *self.running.read().await
982    }
983
984    /// Start listening for events
985    ///
986    /// Spawns a background task that continuously polls for new events
987    /// and broadcasts them to subscribers.
988    ///
989    /// # Returns
990    ///
991    /// Result indicating success or error
992    pub async fn start(&self) -> Result<()> {
993        let mut running = self.running.write().await;
994        if *running {
995            return Ok(()); // Already running
996        }
997        *running = true;
998        drop(running);
999
1000        match self.config.mode {
1001            ConnectionMode::Polling => self.start_polling().await,
1002            ConnectionMode::WebSocket => {
1003                Err(MessageError::Xrpc("WebSocket mode not yet implemented".to_string()))
1004            }
1005        }
1006    }
1007
1008    /// Start polling-based event listening
1009    async fn start_polling(&self) -> Result<()> {
1010        let client = Arc::clone(&self.client);
1011        let running = Arc::clone(&self.running);
1012        let last_cursor = Arc::clone(&self.last_cursor);
1013        let event_tx = self.event_tx.clone();
1014        let poll_interval = self.config.poll_interval;
1015
1016        tokio::spawn(async move {
1017            while *running.read().await {
1018                // Poll for new messages
1019                if let Ok(events) = Self::poll_events(&client, &last_cursor).await {
1020                    for event in events {
1021                        let _ = event_tx.send(event);
1022                    }
1023                }
1024
1025                tokio::time::sleep(poll_interval).await;
1026            }
1027        });
1028
1029        Ok(())
1030    }
1031
1032    /// Poll for new events
1033    async fn poll_events(
1034        client: &Arc<RwLock<XrpcClient>>,
1035        last_cursor: &Arc<RwLock<Option<String>>>,
1036    ) -> Result<Vec<MessageEvent>> {
1037        let cursor = last_cursor.read().await.clone();
1038
1039        let client_guard = client.read().await;
1040        let mut request = XrpcRequest::query("chat.bsky.convo.listConvos");
1041
1042        if let Some(cursor_val) = cursor {
1043            request = request.param("cursor", cursor_val);
1044        }
1045
1046        #[derive(Deserialize)]
1047        #[serde(rename_all = "camelCase")]
1048        struct ConvoListResponse {
1049            convos: Vec<Conversation>,
1050            cursor: Option<String>,
1051        }
1052
1053        let response = client_guard
1054            .query::<ConvoListResponse>(request)
1055            .await
1056            .map_err(|e| MessageError::Xrpc(e.to_string()))?;
1057
1058        // Update cursor for next poll
1059        if let Some(new_cursor) = response.data.cursor {
1060            *last_cursor.write().await = Some(new_cursor);
1061        }
1062
1063        // Convert conversations to events
1064        let mut events = Vec::new();
1065        for convo in response.data.convos {
1066            // Check if this is a new conversation or updated one
1067            if let Some(last_message) = convo.last_message {
1068                let event = MessageEvent::new(
1069                    MessageEventType::NewMessage,
1070                    convo.id.clone(),
1071                )
1072                .with_message(Message {
1073                    id: last_message.id,
1074                    text: last_message.text,
1075                    sender: MessageSender {
1076                        did: last_message.sender.did,
1077                        handle: last_message.sender.handle,
1078                        display_name: last_message.sender.display_name,
1079                        avatar: last_message.sender.avatar,
1080                    },
1081                    sent_at: last_message.sent_at,
1082                    is_from_self: None,
1083                });
1084                events.push(event);
1085            }
1086        }
1087
1088        Ok(events)
1089    }
1090
1091    /// Stop listening for events
1092    ///
1093    /// # Returns
1094    ///
1095    /// Result indicating success or error
1096    pub async fn stop(&self) -> Result<()> {
1097        let mut running = self.running.write().await;
1098        *running = false;
1099        Ok(())
1100    }
1101
1102    /// Reset the polling cursor
1103    ///
1104    /// This will cause the next poll to fetch from the beginning
1105    pub async fn reset_cursor(&self) {
1106        *self.last_cursor.write().await = None;
1107    }
1108}

1169            }
1170            MessageEventType::TypingIndicator => {
1171                // Handle typing indicator (UI concern, no state change needed)
