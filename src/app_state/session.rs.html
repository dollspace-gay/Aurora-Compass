15/// Session-related errors
16#[derive(Debug, thiserror::Error)]
17pub enum SessionStateError {
18    /// Session manager error
19    #[error("Session manager error: {0}")]
20    SessionManager(#[from] SessionManagerError),
21
22    /// Query error
23    #[error("Query error: {0}")]
24    Query(#[from] QueryError),
25
26    /// No current session
27    #[error("No current session")]
28    NoCurrentSession,
29
30    /// Account not found
31    #[error("Account not found: {0}")]
32    AccountNotFound(String),
33}
34
35/// Result type for session state operations
36pub type Result<T> = std::result::Result<T, SessionStateError>;

56/// Query for the current session
57///
58/// This query provides reactive access to the current active session.
59/// It will automatically update when the session changes.
60#[derive(Clone)]
61pub struct CurrentSessionQuery {
62    session_manager: Arc<RwLock<SessionManager>>,
63}
64
65impl CurrentSessionQuery {
66    /// Create a new current session query
67    pub fn new(session_manager: Arc<RwLock<SessionManager>>) -> Self {
68        Self { session_manager }
69    }
70}
71
72#[async_trait]
73impl Query for CurrentSessionQuery {
74    type Data = Option<CurrentSession>;
75
76    async fn fetch(&self) -> crate::query::Result<Self::Data> {
77        let manager = self.session_manager.read().await;
78
79        if let Some(account) = manager.current_account() {
80            Ok(Some(CurrentSession {
81                account: account.clone(),
82                is_active: manager.current_agent().is_some(),
83            }))
84        } else {
85            Ok(None)
86        }
87    }
88
89    fn key(&self) -> QueryKey {
90        QueryKey::new("session", "current")
91    }
92
93    fn config(&self) -> QueryConfig {
94        QueryConfig {
95            stale_time: Duration::from_secs(0), // Always fresh
96            cache_time: Duration::from_secs(300),
97            refetch_on_stale: false, // Manual invalidation only
98            retry: false,
99            retry_count: 0,
100            retry_delay: Duration::from_secs(0),
101        }
102    }
103}

149/// Mutation for switching accounts
150///
151/// This mutation switches the active account and invalidates related queries.
152pub struct SwitchAccountMutation {
153    session_manager: Arc<RwLock<SessionManager>>,
154    query_client: QueryClient,
155}
156
157impl SwitchAccountMutation {
158    /// Create a new switch account mutation
159    pub fn new(session_manager: Arc<RwLock<SessionManager>>, query_client: QueryClient) -> Self {
160        Self { session_manager, query_client }
161    }
162
163    /// Execute the mutation to switch accounts
164    ///
165    /// # Arguments
166    ///
167    /// * `did` - The DID of the account to switch to
168    ///
169    /// # Returns
170    ///
171    /// The new current session
172    pub async fn execute(&self, did: &str) -> Result<CurrentSession> {
173        // Perform the account switch
174        {
175            let mut manager = self.session_manager.write().await;
176            manager.switch_account(did).await?;
177        }
178
179        // Invalidate session-related queries
180        self.invalidate_session_queries().await?;
181
182        // Fetch and return the new current session
183        let query = CurrentSessionQuery::new(Arc::clone(&self.session_manager));
184        let session = self.query_client.fetch(&query).await?;
185
186        session.ok_or(SessionStateError::NoCurrentSession)
187    }
188
189    /// Invalidate all session-related queries
190    async fn invalidate_session_queries(&self) -> Result<()> {
191        self.query_client.invalidate_scope("session").await?;
192        Ok(())
193    }
194}

214        {
215            let mut manager = self.session_manager.write().await;
216            manager.add_account(account).await?;

224        Ok(())
225    }

261/// Session state manager
262///
263/// This provides a high-level API for managing session state reactively.
264#[derive(Clone)]
265pub struct SessionState {
266    session_manager: Arc<RwLock<SessionManager>>,
267    query_client: QueryClient,
268}
269
270impl SessionState {
271    /// Create a new session state manager
272    ///
273    /// # Arguments
274    ///
275    /// * `session_manager` - The underlying session manager
276    /// * `query_client` - The query client for reactive queries
277    pub fn new(session_manager: Arc<RwLock<SessionManager>>, query_client: QueryClient) -> Self {
278        Self { session_manager, query_client }
279    }
280
281    /// Get the current session query
282    pub fn current_session_query(&self) -> CurrentSessionQuery {
283        CurrentSessionQuery::new(Arc::clone(&self.session_manager))
284    }
285
286    /// Get the accounts query
287    pub fn accounts_query(&self) -> AccountsQuery {
288        AccountsQuery::new(Arc::clone(&self.session_manager))
289    }
290
291    /// Get the switch account mutation
292    pub fn switch_account_mutation(&self) -> SwitchAccountMutation {
293        SwitchAccountMutation::new(Arc::clone(&self.session_manager), self.query_client.clone())
294    }
295
296    /// Get the add account mutation
297    pub fn add_account_mutation(&self) -> AddAccountMutation {
298        AddAccountMutation::new(Arc::clone(&self.session_manager), self.query_client.clone())
299    }
300
301    /// Get the remove account mutation
302    pub fn remove_account_mutation(&self) -> RemoveAccountMutation {
303        RemoveAccountMutation::new(Arc::clone(&self.session_manager), self.query_client.clone())
304    }
305
306    /// Get the current session data
307    ///
308    /// This fetches fresh data and caches it.
309    pub async fn get_current_session(&self) -> Result<Option<CurrentSession>> {
310        let query = self.current_session_query();
311        Ok(self.query_client.get(&query).await?)
312    }
313
314    /// Get all accounts
315    ///
316    /// This fetches fresh data and caches it.
317    pub async fn get_accounts(&self) -> Result<AccountList> {
318        let query = self.accounts_query();
319        Ok(self.query_client.get(&query).await?)
320    }
321
322    /// Switch to a different account
323    ///
324    /// # Arguments
325    ///
326    /// * `did` - The DID of the account to switch to
327    pub async fn switch_account(&self, did: &str) -> Result<CurrentSession> {
328        let mutation = self.switch_account_mutation();
329        mutation.execute(did).await
330    }
331
332    /// Add a new account
333    ///
334    /// # Arguments
335    ///
336    /// * `account` - The account to add
337    pub async fn add_account(&self, account: SessionAccount) -> Result<()> {
338        let mutation = self.add_account_mutation();
339        mutation.execute(account).await
340    }
341
342    /// Remove an account
343    ///
344    /// # Arguments
345    ///
346    /// * `did` - The DID of the account to remove
347    pub async fn remove_account(&self, did: &str) -> Result<()> {
348        let mutation = self.remove_account_mutation();
349        mutation.execute(did).await
350    }
351
352    /// Invalidate all session queries
353    ///
354    /// This forces a refetch on the next query.
355    pub async fn invalidate(&self) -> Result<()> {
356        self.query_client.invalidate_scope("session").await?;
357        Ok(())
358    }
