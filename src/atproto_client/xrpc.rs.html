17/// XRPC error with HTTP status and message
18///
19/// This represents errors returned from XRPC endpoints, including both
20/// network failures and application-level errors.
21///
22/// # Examples
23/// ```
24/// use atproto_client::xrpc::XrpcError;
25///
26/// let error = XrpcError::new(404, "NotFound", "Record not found");
27/// assert_eq!(error.status(), 404);
28/// assert!(!error.is_network_error());
29/// ```
30#[derive(Debug, Clone, PartialEq, Eq)]
31pub struct XrpcError {
32    /// HTTP status code
33    status: u16,
34    /// Error code (e.g., "InvalidRequest", "NotFound")
35    error: String,
36    /// Human-readable error message
37    message: String,
38}
39
40impl XrpcError {
41    /// Create a new XRPC error
42    pub fn new(status: u16, error: impl Into<String>, message: impl Into<String>) -> Self {
43        Self {
44            status,
45            error: error.into(),
46            message: message.into(),
47        }
48    }
49
50    /// Get the HTTP status code
51    pub fn status(&self) -> u16 {
52        self.status
53    }
54
55    /// Get the error code
56    pub fn error(&self) -> &str {
57        &self.error
58    }
59
60    /// Get the error message
61    pub fn message(&self) -> &str {
62        &self.message
63    }
64
65    /// Check if this is a network-related error that should be retried
66    ///
67    /// Based on original-bluesky/src/state/messages/convo/const.ts
68    /// Network failure statuses: 1, 408, 425, 429, 500, 502, 503, 504, 522, 524
69    pub fn is_network_error(&self) -> bool {
70        matches!(self.status, 1 | 408 | 425 | 429 | 500 | 502 | 503 | 504 | 522 | 524)
71    }
72
73    /// Check if this error is recoverable (can be retried)
74    pub fn is_recoverable(&self) -> bool {
75        self.is_network_error()
76    }
77}
78
79impl std::fmt::Display for XrpcError {
80    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
81        write!(f, "XRPC error {}: {} - {}", self.status, self.error, self.message)
82    }
83}

91/// HTTP method for XRPC requests
92#[derive(Debug, Clone, Copy, PartialEq, Eq)]
93pub enum HttpMethod {
94    /// GET request (used for queries)
95    Get,
96    /// POST request (used for procedures)
97    Post,
98    /// PUT request
99    Put,
100    /// DELETE request
101    Delete,
102}
103
104impl HttpMethod {
105    /// Convert to reqwest Method
106    pub fn as_str(&self) -> &'static str {
107        match self {
108            HttpMethod::Get => "GET",
109            HttpMethod::Post => "POST",
110            HttpMethod::Put => "PUT",
111            HttpMethod::Delete => "DELETE",
112        }
113    }
114}

116/// XRPC request parameters
117///
118/// Represents a request to an XRPC endpoint with method, path, headers,
119/// query parameters, and optional body.
120#[derive(Debug, Clone)]
121pub struct XrpcRequest {
122    /// HTTP method
123    pub method: HttpMethod,
124    /// NSID path (e.g., "com.atproto.repo.getRecord")
125    pub nsid: String,
126    /// Query parameters
127    pub params: HashMap<String, String>,
128    /// Request headers
129    pub headers: HashMap<String, String>,
130    /// Request body (for POST/PUT)
131    pub body: Option<Vec<u8>>,
132    /// Encoding type (e.g., "application/json")
133    pub encoding: Option<String>,
134}
135
136impl XrpcRequest {
137    /// Create a new GET request (query)
138    pub fn query(nsid: impl Into<String>) -> Self {
139        Self {
140            method: HttpMethod::Get,
141            nsid: nsid.into(),
142            params: HashMap::new(),
143            headers: HashMap::new(),
144            body: None,
145            encoding: None,
146        }
147    }
148
149    /// Create a new POST request (procedure)
150    pub fn procedure(nsid: impl Into<String>) -> Self {
151        Self {
152            method: HttpMethod::Post,
153            nsid: nsid.into(),
154            params: HashMap::new(),
155            headers: HashMap::new(),
156            body: None,
157            encoding: Some("application/json".to_string()),
158        }
159    }
160
161    /// Add a query parameter
162    pub fn param(mut self, key: impl Into<String>, value: impl Into<String>) -> Self {
163        self.params.insert(key.into(), value.into());
164        self
165    }
166
167    /// Add a header
168    pub fn header(mut self, key: impl Into<String>, value: impl Into<String>) -> Self {
169        self.headers.insert(key.into(), value.into());
170        self
171    }
172
173    /// Set the request body (for POST/PUT)
174    pub fn body(mut self, body: Vec<u8>) -> Self {
175        self.body = Some(body);
176        self
177    }
178
179    /// Set the request body from JSON
180    pub fn json_body<T: Serialize>(mut self, value: &T) -> Result<Self, serde_json::Error> {
181        let body = serde_json::to_vec(value)?;
182        self.body = Some(body);
183        self.encoding = Some("application/json".to_string());
184        Ok(self)
185    }
186
187    /// Set encoding type
188    pub fn encoding(mut self, encoding: impl Into<String>) -> Self {
189        self.encoding = Some(encoding.into());
190        self
191    }
192}

198/// XRPC response
199///
200/// Generic response from an XRPC endpoint with headers and data.
201#[derive(Debug, Clone)]
202pub struct XrpcResponse<T> {
203    /// HTTP status code
204    pub status: u16,
205    /// Response headers
206    pub headers: HashMap<String, String>,
207    /// Response data
208    pub data: T,
209}
210
211impl<T> XrpcResponse<T> {
212    /// Create a new response
213    pub fn new(status: u16, headers: HashMap<String, String>, data: T) -> Self {
214        Self { status, headers, data }
215    }
216
217    /// Get a header value
218    pub fn header(&self, key: &str) -> Option<&String> {
219        self.headers.get(key)
220    }
221
222    /// Check if the response is successful (2xx status)
223    pub fn is_success(&self) -> bool {
224        (200..300).contains(&self.status)
225    }
226}

419use std::future::Future;
420use tokio::time::sleep;
421
422/// Configuration for retry behavior
423#[derive(Debug, Clone)]
424pub struct RetryConfig {
425    /// Maximum number of retry attempts
426    pub max_retries: usize,
427    /// Initial delay between retries
428    pub initial_delay: Duration,
429    /// Maximum delay between retries
430    pub max_delay: Duration,
431    /// Backoff multiplier (e.g., 2.0 for exponential backoff)
432    pub backoff_multiplier: f64,
433}
434
435impl Default for RetryConfig {
436    fn default() -> Self {
437        Self {
438            max_retries: 3,
439            initial_delay: Duration::from_millis(100),
440            max_delay: Duration::from_secs(30),
441            backoff_multiplier: 2.0,
442        }
443    }
444}
445
446impl RetryConfig {
447    /// Create a new retry configuration
448    pub fn new(max_retries: usize) -> Self {
449        Self { max_retries, ..Default::default() }
450    }
451
452    /// Set the initial delay
453    pub fn with_initial_delay(mut self, delay: Duration) -> Self {
454        self.initial_delay = delay;
455        self
456    }
457
458    /// Set the maximum delay
459    pub fn with_max_delay(mut self, delay: Duration) -> Self {
460        self.max_delay = delay;
461        self
462    }
463
464    /// Set the backoff multiplier
465    pub fn with_backoff_multiplier(mut self, multiplier: f64) -> Self {
466        self.backoff_multiplier = multiplier;
467        self
468    }
469
470    /// Calculate the delay for a given retry attempt
471    fn calculate_delay(&self, attempt: usize) -> Duration {
472        let delay_ms =
473            self.initial_delay.as_millis() as f64 * self.backoff_multiplier.powi(attempt as i32);
474
475        let delay = Duration::from_millis(delay_ms as u64);
476
477        // Cap at max_delay
478        if delay > self.max_delay {
479            self.max_delay
480        } else {
481            delay
482        }
483    }
484}

486/// Retry an async operation with a configurable retry policy
487///
488/// Reference: original-bluesky/src/lib/async/retry.ts
489///
490/// # Arguments
491/// * `config` - Retry configuration
492/// * `should_retry` - Function to determine if an error should be retried
493/// * `operation` - The async operation to retry
494///
495/// # Examples
496/// ```
497/// use atproto_client::xrpc::{retry, RetryConfig, XrpcError};
498///
499/// async fn example() -> Result<String, XrpcError> {
500///     let config = RetryConfig::new(3);
501///
502///     retry(
503///         config,
504///         |err: &XrpcError| err.is_network_error(),
505///         || async {
506///             // Your operation here
507///             Ok("success".to_string())
508///         }
509///     ).await
510/// }
511/// ```
512pub async fn retry<F, Fut, T, E>(
513    config: RetryConfig,
514    should_retry: impl Fn(&E) -> bool,
515    mut operation: F,
516) -> Result<T, E>
517where
518    F: FnMut() -> Fut,
519    Fut: Future<Output = Result<T, E>>,
520{
521    let mut attempts = 0;
522    let mut last_error: Option<E> = None;
523
524    loop {
525        match operation().await {
526            Ok(result) => return Ok(result),
527            Err(err) => {
528                attempts += 1;
529
530                // Check if we should retry this error
531                if !should_retry(&err) {
532                    return Err(err);
533                }
534
535                // Check if we've exhausted retries
536                if attempts > config.max_retries {
537                    return Err(last_error.unwrap_or(err));
538                }
539
540                // Calculate delay and sleep
541                let delay = config.calculate_delay(attempts - 1);
542                sleep(delay).await;
543
544                last_error = Some(err);
545            }
546        }
547    }
548}

550/// Convenience function to retry network errors
551///
552/// Reference: original-bluesky/src/lib/async/retry.ts:networkRetry
553///
554/// # Examples
555/// ```
556/// use atproto_client::xrpc::{network_retry, XrpcError};
557///
558/// async fn example() -> Result<String, XrpcError> {
559///     network_retry(2, || async {
560///         // Your network operation here
561///         Ok("success".to_string())
562///     }).await
563/// }
564/// ```
565pub async fn network_retry<F, Fut, T>(max_retries: usize, operation: F) -> Result<T, XrpcError>
566where
567    F: FnMut() -> Fut,
568    Fut: Future<Output = Result<T, XrpcError>>,
569{
570    let config = RetryConfig::new(max_retries);
571    retry(config, |err: &XrpcError| err.is_network_error(), operation).await
572}

763///     // Make a query request
764///     let request = XrpcRequest::query("com.atproto.server.describeServer");
765///     let response = client.query::<serde_json::Value>(request).await?;
766///
767///     Ok(())
768/// }
769/// ```
770#[derive(Debug, Clone)]
771pub struct XrpcClient {
772    /// HTTP client
773    client: ReqwestClient,
774    /// Configuration
775    config: XrpcClientConfig,
776}

778impl XrpcClient {
779    /// Create a new XRPC client
780    pub fn new(config: XrpcClientConfig) -> Self {
781        let client = ReqwestClient::builder()
782            .timeout(config.timeout)
783            .user_agent(&config.user_agent)
784            .build()
785            .expect("Failed to build HTTP client");
786
787        Self { client, config }

790    /// Make a query request (GET)
791    ///
792    /// # Examples
793    /// ```
794    /// use atproto_client::xrpc::{XrpcClient, XrpcClientConfig, XrpcRequest};
795    /// use serde::Deserialize;
796    ///
797    /// #[derive(Deserialize)]
798    /// struct ServerDescription {
799    ///     did: String,
800    /// }
801    ///
802    /// async fn example() -> Result<(), Box<dyn std::error::Error>> {
803    ///     let config = XrpcClientConfig::new("https://bsky.social");
804    ///     let client = XrpcClient::new(config);
805    ///
806    ///     let request = XrpcRequest::query("com.atproto.server.describeServer");
807    ///     let response = client.query::<ServerDescription>(request).await?;
808    ///
809    ///     println!("Server DID: {}", response.data.did);
810    ///     Ok(())
811    /// }
812    /// ```
813    pub async fn query<T>(&self, request: XrpcRequest) -> Result<XrpcResponse<T>, XrpcError>
814    where
815        T: for<'de> Deserialize<'de>,
816    {
817        self.execute_request(request).await
818    }

889    /// Set or update the authorization header
890    ///
891    /// This updates the default Authorization header for all subsequent requests.
892    /// Pass `None` to remove the authorization header.
893    ///
894    /// # Example
895    ///
896    /// ```rust
897    /// use atproto_client::xrpc::{XrpcClient, XrpcClientConfig};
898    ///
899    /// let mut config = XrpcClientConfig::new("https://bsky.social");
900    /// let mut client = XrpcClient::new(config);
901    ///
902    /// // Set auth header
903    /// client.set_auth_header(Some("Bearer token123".to_string()));
904    ///
905    /// // Clear auth header
906    /// client.set_auth_header(None);
907    /// ```
908    pub fn set_auth_header(&mut self, auth: Option<String>) {
909        if let Some(token) = auth {
910            self.config
911                .default_headers
912                .insert("Authorization".to_string(), token);
913        } else {
914            self.config.default_headers.remove("Authorization");
915        }
916    }
917
918    /// Execute an XRPC request
919    async fn execute_request<T>(&self, request: XrpcRequest) -> Result<XrpcResponse<T>, XrpcError>
920    where
921        T: for<'de> Deserialize<'de>,
922    {
923        // Build URL
924        let url = format!("{}/xrpc/{}", self.config.service_url, request.nsid);
925
926        // Build reqwest request
927        let mut req = match request.method {
928            HttpMethod::Get => self.client.get(&url),
929            HttpMethod::Post => self.client.post(&url),
930            HttpMethod::Put => self.client.put(&url),
931            HttpMethod::Delete => self.client.delete(&url),
932        };
933
934        // Add query parameters
935        for (key, value) in &request.params {
936            req = req.query(&[(key, value)]);
937        }
938
939        // Add default headers
940        for (key, value) in &self.config.default_headers {
941            req = req.header(key, value);
942        }
943
944        // Add request headers
945        for (key, value) in &request.headers {
946            req = req.header(key, value);
947        }
948
949        // Add body if present
950        if let Some(body) = &request.body {
951            if let Some(encoding) = &request.encoding {
952                req = req.header("Content-Type", encoding);
953            }
954            req = req.body(body.clone());
955        }
956
957        // Execute request
958        let response = req
959            .send()
960            .await
961            .map_err(|e| XrpcError::new(0, "NetworkError", format!("Request failed: {}", e)))?;
962
963        // Convert to XrpcResponse
964        self.parse_response(response).await
965    }
966
967    /// Parse reqwest response into XrpcResponse
968    async fn parse_response<T>(
969        &self,
970        response: ReqwestResponse,
971    ) -> Result<XrpcResponse<T>, XrpcError>
972    where
973        T: for<'de> Deserialize<'de>,
974    {
975        let status = response.status().as_u16();
976
977        // Extract headers
978        let mut headers = HashMap::new();
979        for (key, value) in response.headers() {
980            if let Ok(value_str) = value.to_str() {
981                headers.insert(key.to_string(), value_str.to_string());
982            }
983        }
984
985        // Check if response is an error
986        if !response.status().is_success() {
987            let error_body = response.text().await.unwrap_or_default();
988
989            // Try to parse as XrpcErrorResponse
990            if let Ok(error_response) = serde_json::from_str::<XrpcErrorResponse>(&error_body) {
991                return Err(XrpcError::new(status, error_response.error, error_response.message));
992            } else {
993                return Err(XrpcError::new(
994                    status,
995                    "Unknown",
996                    format!("HTTP {}: {}", status, error_body),
997                ));
998            }
999        }
1000
1001        // Parse successful response
1002        let body = response.text().await.map_err(|e| {
1003            XrpcError::new(0, "ParseError", format!("Failed to read response: {}", e))
1004        })?;
1005
1006        let data: T = serde_json::from_str(&body)
1007            .map_err(|e| XrpcError::new(0, "ParseError", format!("Failed to parse JSON: {}", e)))?;
1008
1009        Ok(XrpcResponse::new(status, headers, data))
1010    }

1013    pub fn config(&self) -> &XrpcClientConfig {
1014        &self.config
1015    }

1017    /// Get the service URL
1018    pub fn service_url(&self) -> &str {
1019        &self.config.service_url
1020    }
